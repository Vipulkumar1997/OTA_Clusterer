#4g ram is necessary to run fess
$ docker run -d -p 8080:8080 -p 9200:9200 --name fess codelibs/fess:latest

#run kibana and expose port to host and connect to fess container
docker run -d -p 5601:5601 -e ELASTICSEARCH_URL=http://172.17.0.2:9200 --name kibana kibana:latest

#assign new port mapping to existing container
$ docker stop fess
$ docker commit fess fess01
$ docker run -d -p 8080:8080 -p 9200:9200 --name fess fess01

#
Go to kibana home http://localhost:5601/.
Input "fess_log" to the textbox of Index name or pattern.
Set "requestedAt" to the Time-field name and Create.
From the top of page, click Settings.
Click Objects.
Click Import and select "fess_log.json" to import example settings.
Click Dashboard.
Click Load Saved Dashboard and select "fess_log" dashboard.
(Change the period from upper right if you want to do.)
